Contexto

Para este ejercicio usaremos los datos de medidas anatmicas recolectados por Heinz et al.(2003) que ya conocimos en el ejercicio prctico anterior (disponibles en el archivo EP09 Datos.csv). Como en este case se requiere de una variable dicotmica, vamos a realizar lo siguiente:

  1. El equipo crea la variable IMC (ndice de masa corporal) como el peso de una persona (en kilogramos) dividida por el cuadrado de su estatura (en metros).
  2. Si bien esta variable se usa para clasificar a las personas en varias clases de estado nutricional (bajo peso, normal, sobrepeso, obesidad, obesidad mrbida), para efectos de este ejercicio, usaremos dos clases: sobrepeso (IMC 23,2) y no sobrepeso (IMC < 23,2).
  3. El equipo crea la variable dicotmica EN (estado nutricional) de acuerdo al valor de IMC de cada persona.

  1. Informacin incompleta:

train_logit <- function(formula, data, folds = 10, positive = "Sobrepeso", threshold = 0.5) {
  set.seed(4263)
  n <- nrow(data)
  indices <- sample(rep(1:folds, length.out = n))
  resultados_cv <- data.frame(
    Fold = integer(),
    Accuracy = numeric(),
    Sensitivity = numeric(),
    Specificity = numeric()
  )
  
  for (k in 1:folds) {
    test_idx <- which(indices == k)
    train_idx <- setdiff(1:n, test_idx)
    modelo_cv <- glm(formula, data = data[train_idx, ], family = binomial(link = "logit"))
    probas <- predict(modelo_cv, newdata = data[test_idx, ], type = "response")
    clases <- unique(data[[as.character(formula[[2]])]])
    clase_neg <- setdiff(clases, positive)
    pred <- ifelse(probas > threshold, positive, clase_neg)
    real <- data[test_idx, as.character(formula[[2]])]
    
    TP <- sum(pred == positive & real == positive)
    TN <- sum(pred == clase_neg & real == clase_neg)
    FP <- sum(pred == positive & real == clase_neg)
    FN <- sum(pred == clase_neg & real == positive)
    
    sensibilidad <- ifelse((TP + FN) == 0, NA, TP / (TP + FN))
    especificidad <- ifelse((TN + FP) == 0, NA, TN / (TN + FP))
    exactitud <- mean(pred == real)
    
    resultados_cv <- rbind(resultados_cv, data.frame(
      Fold = k,
      Accuracy = exactitud,
      Sensitivity = sensibilidad,
      Specificity = especificidad
    ))
  }
  
  modelo <- glm(formula, data = data, family = binomial(link = "logit"))
  resumen <- summary(modelo)
  probas_full <- predict(modelo, newdata = data, type = "response")
  clases <- unique(data[[as.character(formula[[2]])]])
  clase_neg <- setdiff(clases, positive)
  pred_full <- ifelse(probas_full > threshold, positive, clase_neg)
  real_full <- data[[as.character(formula[[2]])]]
  
  TP <- sum(pred_full == positive & real_full == positive)
  TN <- sum(pred_full == clase_neg & real_full == clase_neg)
  FP <- sum(pred_full == positive & real_full == clase_neg)
  FN <- sum(pred_full == clase_neg & real_full == positive)
  sensibilidad <- ifelse((TP + FN) == 0, NA, TP / (TP + FN))
  especificidad <- ifelse((TN + FP) == 0, NA, TN / (TN + FP))
  exactitud <- mean(pred_full == real_full)
  
  list(
    finalModel = modelo,
    resumen_modelo = resumen,
    resultados_cv = resultados_cv,
    sensibilidad = sensibilidad,
    especificidad = especificidad,
    exactitud = exactitud
  )
}

# calculo de matriz de confusion
calcular_metricas <- function(pred, real, positive = "Sobrepeso") {
  pred <- factor(pred, levels = c("No_sobrepeso", "Sobrepeso"))
  real <- factor(real, levels = c("No_sobrepeso", "Sobrepeso"))
  
  TP <- sum(pred == positive & real == positive)
  TN <- sum(pred != positive & real != positive)
  FP <- sum(pred == positive & real != positive)
  FN <- sum(pred != positive & real == positive)
  
  exactitud <- mean(pred == real)
  sensibilidad <- ifelse((TP + FN) == 0, NA, TP / (TP + FN))
  especificidad <- ifelse((TN + FP) == 0, NA, TN / (TN + FP))
  
  matriz <- matrix(c(TN, FP, FN, TP), nrow = 2, byrow = TRUE,
                   dimnames = list("Real" = c("No_sobrepeso", "Sobrepeso"),
                                   "Predicho" = c("No_sobrepeso", "Sobrepeso")))
  
  list(
    tabla = matriz,
    exactitud = exactitud,
    sensibilidad = sensibilidad,
    especificidad = especificidad
  )
}
